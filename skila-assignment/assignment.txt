-- SQL200 Worksheet (Sakila Database)

-- Q1: Films with rental rate 9.99 or 4.99
SELECT film_id, title, rental_rate
FROM film
WHERE rental_rate = 9.99 OR rental_rate = 4.99;



-- Q2: Films between 90–120 mins and rating PG / PG-13
SELECT title, length, rating
FROM film
WHERE length BETWEEN 90 AND 120
AND (rating = 'PG' OR rating = 'PG-13');



-- Q3: Actors whose last name starts with S or ends with N
SELECT actor_id, first_name, last_name
FROM actor
WHERE last_name LIKE 'S%' OR last_name LIKE '%N';

-- Q4: Active customers with email ending .org or .net
SELECT customer_id, first_name, last_name, email
FROM customer
WHERE active = 1
AND (email LIKE '%.org' OR email LIKE '%.net');



-- Q5: Inactive customers from store 1
SELECT customer_id, store_id, active
FROM customer
WHERE store_id = 1 AND active = 0;


-- Q6: Payments between 2–5 in February 2007
SELECT payment_id, customer_id, amount, payment_date
FROM payment
WHERE amount BETWEEN 2 AND 5
AND payment_date >= '2007-02-01'
AND payment_date < '2007-03-01';


-- Q7: Rentals not returned yet
SELECT rental_id, rental_date, return_date, customer_id
FROM rental
WHERE return_date IS NULL;

-- Q8: Addresses in Texas or California with postal code
SELECT address_id, address, district, postal_code
FROM address
WHERE (district = 'Texas' OR district = 'California')
AND postal_code IS NOT NULL;


-- Q9: Films with given replacement cost and no 'A' in title
SELECT film_id, title, replacement_cost
FROM film
WHERE (replacement_cost = 12.99 OR replacement_cost = 16.99 OR replacement_cost = 28.99)
AND title NOT LIKE '%A%';



-- Q10: Inventory condition based on store and film range
SELECT inventory_id, film_id, store_id
FROM inventory
WHERE
(store_id = 1 AND film_id BETWEEN 1 AND 50)
OR
(store_id = 2 AND film_id BETWEEN 51 AND 100);
